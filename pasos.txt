instalar git:
   apt-get intall git

crear carpeta para crear o repositorio local:
   mkdir ~/git_cxt

entrar no directorio creado:
   cd ~/git_cxt

crear o repositorio:
   git init

engadir arquivos:
   todos os arquivos:
      git add .
   arquivo en concreto:
      git add arquivo.algo

introducir nome de usuarios e correo no programa:
   git config --global user.name "daniel"
   git config --global user.mail "daniel@correo.son"

facer commit:
   git commit
